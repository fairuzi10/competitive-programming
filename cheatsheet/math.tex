\subsection{Extended Euclidean}

\begin{cppcode}
	// given a and b, return gcd(a, b) and find x and y which satisfy a*x+b*y = gcd(a, b)
	int gcd (int a, int b, int &x, int &y) {
		if (a == 0) {
			x = 0;
			y = 1;
			return b;
		}
		int x1, y1;
		int d = gcd (b % a, a, x1, y1);
		x = y1 - (b / a) * x1;
		y = x1;
		return d;
	}
\end{cppcode}
